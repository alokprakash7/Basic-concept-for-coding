{
 
   int N;
   cin >>N;
  
   
       for(int i = N; i>=1; i--){
           
           for(int j = 1; j<=N; j++){
               if(j>=i){
                   cout <<j-(i-1);
         }
         else{
             cout <<" ";
         }
           
           }
           cout <<endl;
           
           
           
       }

#include <iostream>

using namespace std;

int main()
{
    int N;
    cin>> N;
    int n = 2*N;
    for(int i=1; i<=n/2; i++){
        for(int j =1; j<=i; j++){
            cout<<"*";
         
        }
        cout<<"\n";
    }
    for(int k = 1; k<=n/2; k++){
        for(int l = n/2; l>=k; l--){
            cout <<"*";
        }
        cout <<"\n";
    }

    return 0;
}
